ON linnad

AS

    USER_NAME(), 
    GETDATE(), 
    'on tehtud UPDATE k채sk',
    CONCAT(
        'vanad andmed - linn: ', deleted.linnanimi,

INNER JOIN inserted ON deleted.linnID = inserted.linnID;
